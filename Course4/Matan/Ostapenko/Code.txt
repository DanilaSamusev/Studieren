# Тест Шапиро–Уилка

#  Нормальное распределение
set.seed(0)
shapiro.test(rnorm(100, mean = 2, sd = 5))

#  Равномерное распределение
set.seed(0)
shapiro.test(runif(100, min = -10, max = 10))

#  Распределение высот деревьев
colnames(trees)
x <- trees[, "Height"]
shapiro.test(x)
